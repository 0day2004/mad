Experiment No: 7
Title: Firebase Authentication using Flutter

Aim:
To implement user authentication using Firebase in a Flutter application.

Requirements:

* Flutter SDK
* VS Code
* Firebase Account
* Android Emulator / Chrome

---

### Step 1: Create Flutter Project

Open VS Code Terminal (Ctrl + `)

Run:

flutter create firebase_app
cd firebase_app
code .

---

### Step 2: Create Firebase Project

1. Open: https://console.firebase.google.com
2. Click "Create Project"
3. Enter name → firebase_app
4. Click Continue → Create

---

### Step 3: Add Android App

1. Click Android icon 🤖
2. Get package name from:

android/app/src/main/AndroidManifest.xml

Example:
com.example.firebase_app

3. Paste in Firebase
4. Click Register

---

### Step 4: Download JSON

Download file:

google-services.json

Place it in:

android/app/google-services.json

---

### Step 5: Enable Authentication

1. Firebase → Authentication
2. Click "Get Started"
3. Go to "Sign-in Method"
4. Enable:
   Email/Password

---

### Step 6: Add Dependencies

Open pubspec.yaml

Add:

dependencies:
flutter:
sdk: flutter
firebase_core: ^2.25.4
firebase_auth: ^4.17.4

Run in terminal:

flutter pub get

---

### Step 7: Android Configuration

Open:
android/build.gradle

Add:

classpath 'com.google.gms:google-services:4.3.15'

---

Open:
android/app/build.gradle

Add at bottom:

apply plugin: 'com.google.gms.google-services'

---

### Step 8: Write Code

Open:
lib/main.dart

Paste:

import 'package:flutter/material.dart';
import 'package:firebase_core/firebase_core.dart';
import 'package:firebase_auth/firebase_auth.dart';

void main() async {
WidgetsFlutterBinding.ensureInitialized();
await Firebase.initializeApp();
runApp(const MyApp());
}

class MyApp extends StatelessWidget {
const MyApp({super.key});

@override
Widget build(BuildContext context) {
return const MaterialApp(
debugShowCheckedModeBanner: false,
home: LoginPage(),
);
}
}

class LoginPage extends StatefulWidget {
const LoginPage({super.key});

@override
State<LoginPage> createState() => _LoginPageState();
}

class _LoginPageState extends State<LoginPage> {
final email = TextEditingController();
final pass = TextEditingController();

Future<void> login() async {
try {
await FirebaseAuth.instance.signInWithEmailAndPassword(
email: email.text.trim(),
password: pass.text.trim(),
);

```
  Navigator.push(

  ScaffoldMessenger.of(context).showSnackBar(
    const SnackBar(content: Text("Login Failed")),
  );
}
```

}

@override
Widget build(BuildContext context) {
return Scaffold(
appBar: AppBar(title: const Text("Login")),
body: Padding(
padding: const EdgeInsets.all(20),
child: Column(
children: [
TextField(
controller: email,
decoration: const InputDecoration(labelText: "Email"),
),
TextField(
controller: pass,
obscureText: true,
decoration: const InputDecoration(labelText: "Password"),
),
const SizedBox(height: 20),
ElevatedButton(
onPressed: login,
child: const Text("Login"),
),
],
),
),
);
}
}

class HomePage extends StatelessWidget {
final String user;

const HomePage({super.key, required this.user});

@override
Widget build(BuildContext context) {
return Scaffold(
appBar: AppBar(title: const Text("Home")),
body: Center(
child: Text("Welcome $user"),
),
);
}
}
